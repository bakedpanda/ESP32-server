# USB tool pattern (wraps with SerialLock):
@mcp.tool()

    try:
        with SerialLock(port):
            return module_function(port, ...)
    except TimeoutError as e:
        return {"error": "serial_lock_timeout", "detail": str(e)}

# WiFi tool pattern (no SerialLock):
@mcp.tool()
def some_wifi_tool(host: str, ...) -> dict:
